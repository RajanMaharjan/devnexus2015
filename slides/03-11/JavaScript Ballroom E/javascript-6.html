Let Declaration

  • Declares a variable within the scope of the current block

    • Slightly different than var, which declares a variable within the block of the enclosing function

function letDeclaration(condition) {
    var varValue;
    if (condition) {

        // varValue exists here

        let letValue = 100;
        return letValue;
    }

    // letValue doesn't exist here

    return varValue;
}

Let Declaration

function letFor() {
    for (let x = 0; x++; x < 10) {
        // x is only available here
        console.log(x);
    }
}

Let Declarations

  • Cannot redefine variables in the same scope

var foo = "bar";
let foo = "bar"; // syntax error

Let Declarations

  • Cannot reference before it is defined in the current block

function letReferenceError() {
    console.log(typeof letValue); // undefined
    if (condition) {
        console.log(typeof letValue); // reference error
        let letValue = 100;
        return letValue;
    }
}

Objects

Object.assign()

  • Like a mixin

    • Copies members from supplier to receiver

  • Can assign many suppliers to one object

function TheOriginal() { /*...*/ }
TheOriginal.prototype = {
    constructor: TheOriginal,
    someValue: 1,
    doSomething: function() { /*...*/ }
}

var myObject = {}
Object.assign(myObject, TheOriginal.prototype);

myObject.doSomething();
console.log(myObject.someValue);

Object.observe()

var model = {
    // some data
};

Object.observe(model, function(changes){

    // Aasynchronous callback
    changes.forEach(function(change) {
        console.log(change.type, change.name, change.oldValue);
    });

});

        yield items[i];
    }

console.log(iterator.next());
	// "{ value: 1, done: false }"

Promises

  • Cornerstone of ES6’s async behavior

  • A Promise can be in any one of these states:

    • Pending

    • Fulfilled

    • Rejected

    • Settled

Promises

var thisPromiseCount = ++promiseCount;

 // We make a new promise: we promise the string 'result' (after waiting 3s)
 var p1 = new Promise(
   // The resolver function is called with the ability to resolve or
   // reject the promise
   function(resolve, reject) {
     // This only is an example to create asynchronism
     window.setTimeout(
       function() {
         // We fulfill the promise !
         resolve(thisPromiseCount)
       }, Math.random() * 2000 + 1000);
   });

 // We define what to do when the promise is fulfilled
 p1.then(
   // Just log the message and a value
   function(val) {
     console.log("Promise fulfilled! Value = " + val);
   });

Modules

  • Lack of modules is a pain in the butt, especailly in JavaScript

    • Core feature: encapsulation

      • Functions, classes, variables declared inside a module are scoped to the module

      • Modules can export their members

Modules

  • Two primary solutions exist today

    • CommonJS

      • Concisce syntax

    • Mostly server-side

      • Synchronous

    • Asynchronous Module Definition (AMD)

      • More complicated syntax

      • Mostly client-side

      • Asynchronous

Modules

  • ES6 Modules combine features from both AMD and CommonJS

  • Supports synchronous and asynchronous loading

  • Supports cyclic references

  • Module structure is static

Exports

  • Two kinds of exports:

    • Default (one per module)

      • Favored approach

    • Named (multiple per module)

  • A module can have both named and default exports

Exports

// myModule.js

export var id;
var name;
const MAX_ITEMS = 10000;

function process(...items) {
    // ...
};

export default class Runner {
    function start() { /*...*/ }
    function stop() { /*...*/ }
};

export process;
export name;
export { MAX_ITEMS as max, name };
}

Imports

// Default
import runner from "src/my-module";

// Default plus named
import runner, {process, name} from "src/my-module";

// Just named
import {process, name} from "src/my-module";

Working with Modules Programmatically

System.import('my-module')
    .then(my_module => {
        // Use the module
    }).catch(error => {
        //...
    });

Promise.all(
    ['module1', 'module2', 'module3']
        .map(mod => System.import(mod)))
    .then(([module1, module2, module3]) => {
        // Use module1, module2, module3
    });
